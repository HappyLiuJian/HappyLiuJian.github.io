面向对象

ECMAScript中没有类的概念，因此其对象定义为:一组无序属性的集合，其属性可以包含基本值、对象和函数，可以直接把对象理解为散列表

对象的属性类型：数据属性和访问器属性

数据属性有四个内部特性：

  • [[Configurable]]:能否删除并重新定义
  • [[Enumerable]]:能否通过for-in循环属性
  • [[Write]]:能否修改

  • [[Value]]:这个属性的数据值
    设置内部特性使用方法defineproperty()
    如下：

    var person = {};
    Object.defineProperty(person,”name”,{
    writable: false;
    value: “liujian”
    })
    alert(person.name);
    person.name = “ouou”;
    alert(person.name);

设置person对象的的属性name的特性writable为false，这就不可改写了，所以下一次赋值name就会失败

访问器属性包含一对函数setter和getter。也有四个内部特性“

  • [[Configurable]]:能否删除并重新定义
  • [[Enumerable]]:能否通过for-in循环属性
  • [[Get]]:读取属性时调用的函数，默认为undefined

  • [[Set]]:写入属性时调用的函数，默认为undefined
    设置访问器属性不能直接定义，只能通过defineProperty()方法来定义
    如下：

    var book = {

    _year = 2014;        //定义默认属性_year,一般加_的属性名表示只能通过对象方法访问
edition: 1;            //定义默认属性edition

    }

    Object.defineProperty(book,”year”,{ //设置访问器属性year

    get: function() {
    return this._year;        //设置getter函数，返回_year的值
}
set: function(newValue) {    //设置setter函数，确定正确的edition版本，并且给_year赋予新值
    if (newValue>2014) {
        this._year = newValue;
        this.edition += newValue -2014;
    }
}

    })

定义多个属性

使用方法defineProperties(),如下：

var book = {};

Object.defineProperties(book,{
    _year: {
        value: 2014
    },
    edition: {
        value: 1
    },
    year: {
        get: function() {
            return this._year;
        },
        set: function(newValue) {
            if (newValue > 2014) {
                this._year = newValue;
                this.edition += newValue - 2014;
            }
        }
    }
});

读取属性的特性

使用方法getOwnPropertyDescription(),如下：

var book = {};

Object.defineProperties(book,{
    _year: {
        value: 2014
    },
    edition: {
        value: 1
    },
    year: {
        get: function() {
            return this._year;
        },
        set: function(newValue) {
            if (newValue > 2014) {
                this._year = newValue;
                this.edition += newValue - 2014;
            }
        }
    }
})
var descriptor = Object.getOwnPropertyDescriptor(book, "_year");
alert(descriptor.value);            //2014
alert(descriptor.configurable);        //false
alert(typeof descriptor.get);        //"undefined"

工厂模式创建对象

由于没有接口的概念，所以设计了工厂模式用来创建类：使用函数封装创建接口，如下：

function createPerson(name, age, job) {
    var o = new Object();
    o.name = name;
    o.age = age;
    o.job = job;
    o.seyName = function() {
        alert(this.name);
    }
    return o;
}
var person1 = createPerson("liujian",20,"student");
var person2 = ...

但是工厂模式不能解决对象识别问题

构造函数模式

如下：

function Person(name, age, job) {
    this.name = name;
    this.age = age;
    this.job = job;
    this.seyName = function() {
        alert(this.name);
    }
}

var person1 = Person("liujian",20,"student");
var person2 = ...

构造函数一般以大写字母开头的单词表示，每个对象实例包含一个constructor属性，指向构造函数
构造函数也是函数，也能通过普通方式调用
构造函数的问题在于每一次创建新实例的时候，里面的方法都要重新创建一遍

原型模式

每一个函数都有一个原型属性prototype，这个属性相当于一个指针，指向一个对象。prototype可以看做所有以这个函数创建的对象实例的原型对象，相当于Object()对象。

function Person() {
}

Person.prototype.name = "liujian";
Person.prototype.age = 20;
Person.prototype.job = "student";
Person.prototype.friends = ["huoming","chenpudi"]
Person.prototype.seyName = function() {
    alert("this.name");
};

var person1 = new Person();
person1.sayName();            //"liujian"
var person2 = new Person();
person2.sayName();            //"liujian"
person1.friends.push("jiangchuan");
alert(person2.friends);        //"guoming","chenpudi","jiangchuan"对于内部引用类型来说，第一个实例对其的改动会导致第二个实例中也改动

组合使用构造函数和原型

由于原型模式不能传递参数并且原型模式定义的属性和参数都是所有实例共享的，而构造函数则可以传递参数并且定义私有的属性(对于内部引用类型来说，第一个实例对其的改动会导致第二个实例中也改动)，所以采用二者结合的方法

function Person(naem,age,job) {
    this.name = name;
    this.age = age;
    this.job = job;
    this.friends = ["guoming","chenpudi"];
}

Person.prototype = {
    constructor : Person;
    sayName : function() {
        alert(this.name);
    }
}

var person1 = new Person("liujian","20","student");
var person2 = new Person("jiajia","20","student");

person1.friends.push("jige");
alert(person1.friends);                /"guoming","chenpudi","jige"
alert(person2.friends);                //"guoming","chenpudi"

动态原型模式

把原型封装到构造函数中，但是仅在必要的情况下进行原型的初始化

function Person(naem,age,job) {
    this.name = name;
    this.age = age;
    this.job = job;
    this.friends = ["guoming","chenpudi"];

    if (typeof this.sayName != "function") {        //只在不存在sayName的情况下才添加该方法，即只会在初次运行构造函数的时候运行
        Person.prototype.sayName = function() {
            alert(this.name);
        }
    }
}

寄生构造函数

创建一个函数，函数仅仅用来封装对象代码，表面看非常像构造函数

function Person(name, age, job) {        //和工厂模式的代码几乎相同
    var o = new Object();
    o.name = name;
    o.age = age;
    o.job = job;
    o.seyName = function() {
        alert(this.name);
    }；
    return o;
}

var friend = new Person("liujian",20,"student");
friends.sayName;            //"liujian"

稳妥构造函数

指的是没有公共属性，方法不引用this的稳妥对象的构造，适用于安全环境

function Person(name,age,job) {                        //不使用this指针
    var o = new Object();
    o.sayName = function() {
        alert(name);
    }
    return o;
}

var friend = Person("liujian",20,"student");        //不使用new操作符
friends.sayName;            //"liujian"
